%----------------------------------------------------------
\renewcommand{\cftfigfont}{\fontsize{12}{18}\selectfont}

\renewcommand{\cfttabpagefont}{\fontsize{12}{18}\selectfont}

%----------------------------------------------------------
% Number width
%----------------------------------------------------------
\setlength{\cftfignumwidth}{3.8em}
\setlength{\cfttabnumwidth}{3.8em}

%----------------------------------------------------------
% Space between entries
%----------------------------------------------------------
\setlength{\cftbeforefigskip}{6pt}
\setlength{\cftbeforetabskip}{6pt}

%----------------------------------------------------------
% No dots
%----------------------------------------------------------
\renewcommand{\cftfigleader}{\hfill}
\renewcommand{\cfttableader}{\hfill}

% List of Figures
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\MyListOfFigures}{

\clearpage

\begin{center}
{\fontsize{16}{20}\selectfont\bfseries List of Figures}
\end{center}

\vspace{1cm}

\noindent
\begin{tabular*}{\textwidth}{@{\extracolsep{\fill}}l c r}
\textbf{Figure No.} &
\textbf{Description} &
\textbf{Page}
\end{tabular*}

\vspace{0.4cm}

\begingroup

\renewcommand{\numberline}[1]{##1\hspace{2em}}

\endgroup

}
